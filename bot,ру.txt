import telebot

TOKEN = "8731025825:AAHYTgEN2ps3WORBB1B3qIDzCG5x2zDbg8Q"

SHEET_ID = "1f6WqnXizDjbWkaT_nn2IAhkrnjjIy8mG"
CSV_URL = f"https://docs.google.com/spreadsheets/d/{SHEET_ID}/export?format=csv"

bot = telebot.TeleBot(TOKEN)

def load_data():
    df = pd.read_csv(CSV_URL)
    df = df.fillna("")
    return df

@bot.message_handler(commands=['start'])
def start(message):
    bot.reply_to(message, "👋 Salom! Ism yozing, men telefonni topaman.")

@bot.message_handler(func=lambda message: True)
def search_user(message):
    text = message.text.lower()
    df = load_data()

    for _, row in df.iterrows():
        ism = str(row.iloc[0]).lower()
        telefon = str(row.iloc[1])

        if text in ism:
            bot.reply_to(message, f"📞 {row.iloc[0]}:\n{telefon}")
            return

    bot.reply_to(message, "❌ Topilmadi")
